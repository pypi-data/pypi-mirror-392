# **1. Answer**

```python
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from scipy.stats import zscore

df1 = pd.read_csv("Diabetes.csv")
df2 = pd.read_csv("house_price.csv")

# Outliers in Diabetes dataset
num_cols = df1.select_dtypes(include='number')
z_scores = np.abs(zscore(num_cols))
outliers = np.where(z_scores > 3)
print("Outlier positions (Diabetes):", outliers)

# Bivariate plot (Example: Glucose vs Outcome)
sns.catplot(x="Outcome", y="Glucose", kind="box", data=df1)
plt.title("Glucose vs Outcome")
plt.show()

# Mode imputation for categorical columns (>10% missing)
cat_cols = df2.select_dtypes(exclude='number').columns
for col in cat_cols:
    if df2[col].isnull().mean() > 0.10:
        df2[col].fillna(df2[col].mode()[0], inplace=True)

# Z-score normalization for numerical feature (example: Price)
df2["Price_z"] = zscore(df2["Price"])

sns.scatterplot(x=df2.index, y=df2["Price_z"])
plt.title("Z-score Normalized Price")
plt.show()
```

# **2. Answer**

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import confusion_matrix, accuracy_score, precision_score, recall_score, f1_score

df = pd.read_csv("house_price.csv")

# Drop missing values for simplicity
df = df.dropna()

# Define X and Y
X = df.drop(columns=['Price'])   # Example target
y = (df['Price'] > df['Price'].median()).astype(int)  # Convert price to classification

# Train-test split 75/25
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.25, random_state=42
)

# Different Random Forest Models
models = [
    RandomForestClassifier(n_estimators=50),
    RandomForestClassifier(n_estimators=100, max_depth=5),
    RandomForestClassifier(n_estimators=150, min_samples_split=4),
    RandomForestClassifier(n_estimators=200, max_features='sqrt')
]

best_acc = 0
best_model = None

for m in models:
    m.fit(X_train, y_train)
    pred = m.predict(X_test)
    acc = accuracy_score(y_test, pred)
    print(f"Model: {m} | Accuracy: {acc}")
    if acc > best_acc:
        best_acc = acc
        best_model = m

print("\nBest Model:", best_model)

# Predictions
y_pred = best_model.predict(X_test)

# Confusion Matrix
cm = confusion_matrix(y_test, y_pred)
print("Confusion Matrix:\n", cm)

# Metrics
acc = accuracy_score(y_test, y_pred)
prec = precision_score(y_test, y_pred)
rec = recall_score(y_test, y_pred)
f1 = f1_score(y_test, y_pred)

acc, prec, rec, f1
```
